

\begin{scifitablebox}[screenbox,title=\textcolor{PanelGlow}{Conditions (Quick Reference)}]

\textbf{Blinded}
\begin{itemize}
\item Fails sight-based checks
\item Attacks vs you: advantage; your attacks: disadvantage
\end{itemize}

\textbf{Charmed}
\begin{itemize}
\item Cannot harm charmer
\item Charmer has advantage on social checks
\end{itemize}

\textbf{Frightened}
\begin{itemize}
\item Disadvantage on checks/attacks (while source in sight)
\item Cannot move closer to source
\end{itemize}

\textbf{Frozen}
\begin{itemize}
\item Incapacitated; cannot move/speak
\item Fails Dexterity saves; attacks vs you have advantage
\item Primed (Cold); detonation ends condition
\end{itemize}

\textbf{Grappled}
\begin{itemize}
\item Speed = 0; no speed bonuses
\item Ends if grappler incapacitated or target moved away
\end{itemize}

\textbf{Deafened}
\begin{itemize}
\item Fails hearing-based checks
\end{itemize}

\textbf{Incapacitated}
\begin{itemize}
\item Cannot take actions or reactions
\end{itemize}

\textbf{Exhaustion (6 Levels)}
\begin{enumerate}
\item Disadvantage on ability checks
\item Speed halved
\item Disadvantage on attacks/saves
\item HP max halved
\item Speed = 0
\item Death
\end{enumerate}
\begin{itemize}
\item Effects stack cumulatively
\item Long rest reduces level by 1 (with food/water)
\end{itemize}

\textbf{Indoctrinated (Optional Campaign Condition)}
\begin{itemize}
\item Caused by prolonged exposure to Reaper tech
\item Every 24h: DC 10 Wisdom save (DC increases over time)
\item Failure: gain 1 level of indoctrination
\end{itemize}

\end{scifitablebox}

\begin{scifitablebox}[screenbox,title=\textcolor{PanelGlow}{Types of Damage}]
\begin{itemize}
  \item \textbf{Acid.} The corrosive spray of a chemical and the dissolving enzymes secreted by thresher maw spit deal damage.
  \item \textbf{Bludgeoning.} Blunt force attacks - hammers, falling, constriction, and the like - deal bludgeoning damage.
  \item \textbf{Cold.} The painful chill from a cryo beam breath deal cold damage.
  \item \textbf{Fire.} Many tech spells use plasma and flames to deal fire damage
  \item \textbf{Force.} One of the primary biotic damage types. Mass effect fields that pummel, pierce, or apply pressure deal force damage.
  \item \textbf{Lightning.} Shock damage. Any damage done by an electrical discharge or source. Deals double damage to shields.
  \item \textbf{Necrotic.} The second biotic damage type. Necrotic damage uses dark energy and mass effect fields on a subatomic level.
  \item \textbf{Piercing.} Puncturing and impaling attacks, including bullets and monsters' bites, deal piercing damage.
  \item \textbf{Poison.} Venomous stings, poisoned blades, and the toxic gas deal poison damage.
  \item \textbf{Psychic.} A shock to the nervous system, rather than fear or mental damage. But the latter is possible.
  \item \textbf{Radiant.} Radiation or damage on a nuclear level, rather than divine light.
  \item \textbf{Slashing.} Swords, axes, and monsters' claws deal slashing damage.
  \item \textbf{Thunder.} Shockwaves and explosions. Thunder damage is primarily caused by grenades.
\end{itemize}
\end{scifitablebox}
