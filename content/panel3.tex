\begin{scifitablebox}[screenbox,title=\textcolor{PanelGlow}{Actions}]
\begin{itemize}
    \item \textbf{Attack - \underline{Weapon System}:} \textit{Make a ranged attack roll. Do not add your proficiency bonus unless you are proficient with Starship Systems (Weapons).} 
    \item \textbf{Targeted Attack - \underline{Weapon System}:} \textit{Same as normal attack but: Name a system on target. Target get +5 AC. On hit system is disabled and crew at station takes half damage.}
    \item \textbf{Maneuver - \underline{Helm}:} \textit{DC 15 Vehicle Handling check. Success adds half ship speed to AC until crews mext turn.}
    \item \textbf{Hard Maneuver - \underline{Helm}:} \textit{DC 20 Vehicle Handling check. Success adds double ship speed to AC until crews mext turn. All on ship must make Constitution save\footnote{Personally use DC 20(adjust for ship size) minus how much pilot was over check. EG: pilot rolls 25, making con save 15. Best use is to state this to the pilot when they take action before they roll.}. On fail disadvantage on next action and no move.}
    \item \textbf{Repair - \underline{Anywhere}:} \textit{DC 15 Engineering check. Success restores 50 Hull Points.}
    \item \textbf{Advanced Repair - \underline{Anywhere}:} \textit{DC 20 Engineering check. Success restores 150 Hull Points.}
    \item \textbf{System Repair - \underline{Disabled System}:} \textit{DC 20 Engineering check. Success restores system.}
    \item \textbf{Restore - \underline{Drive}:} \textit{DC 15 Engineering check. Success restores 100 shield.}
    \item \textbf{Harden - \underline{EWS}:} \textit{DC 10 Electronics check. Success until crews next turn all hacking attempt against ship has disadvantage.}
    \item \textbf{Sabotage - \underline{EWS}:} \textit{Electronics check based on table.}
    \item \textbf{Scan - \underline{SSC}:} \textit{DC 15 Investigation check. Success means asking the GM a question about taget ship. The GM must tell the truth.}
    \item \textbf{Flee - \underline{Navigation}:} \textit{DC 14 Science check. Success means moving to another system.}
    \item \textbf{Move - \underline{Anywhere}:} \textit{Move character to other ship system}
    \item \textbf{Help - \underline{Anywhere}:} \textit{Other person at your current location gets advantage}
    \item \textbf{Defer to VI - \underline{Anywhere}:} \textit{Attempt to use a system you are not at. Check is done with a clean 1d20 with no modifiers.}
\end{itemize}
\end{scifitablebox}
