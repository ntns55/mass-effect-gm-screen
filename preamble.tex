% Standard 2-column panel table
\newcommand{\paneltabletwo}[3]{%
  % #1 title
  % #2 column spec, e.g. {>{\RaggedRight\arraybackslash}p{0.7\linewidth} >{\centering\arraybackslash}p{0.18\linewidth}}
  % #3 rows
  \begin{scifitablebox}
  \ifstrempty{#1}{}{%
    {\bfseries\color{PanelGlow}#1}\par\vspace{0.4em}
  }
  \arrayrulecolor{TableRule}
  \rowcolors{2}{TableRowAlt}{PanelBoxBg}
  \begin{tabularx}{\linewidth}{#2}
    \rowcolor{TableHeaderBg}
    #3
  \end{tabularx}
  \end{scifitablebox}
}

% Easy 2-column table wrapper
\newcommand{\simpletwotable}[4]{%
  \begin{scifitablebox}
  \ifstrempty{#1}{}{%
    {\bfseries\color{PanelGlow}#1}\par\vspace{0.4em}
  }
  {\color{PanelText} % 👈 enforce body text color
  \arrayrulecolor{TableRule}
  \rowcolors{2}{TableRowAlt}{PanelBoxBg}
  \begin{tabularx}{\linewidth}{>{\color{PanelText}\RaggedRight\arraybackslash}X >{\color{PanelText}\centering\arraybackslash}p{0.22\linewidth}}
    \rowcolor{TableHeaderBg}
    \tablehead{#2} & \tablehead{#3} \\
    #4
  \end{tabularx}
  }
  \end{scifitablebox}
}
