% Background helper
\newcommand{\panelbackground}[4]{%
  \AddToShipoutPictureBG*{%
    \begin{tikzpicture}[remember picture,overlay]
      \node[anchor=south west] at (current page.south west)
        {\includegraphics[width=.25\paperwidth,height=\paperheight]{#1}};
      \node[anchor=south west] at ([xshift=.25\paperwidth]current page.south west)
        {\includegraphics[width=.25\paperwidth,height=\paperheight]{#2}};
      \node[anchor=south west] at ([xshift=.50\paperwidth]current page.south west)
        {\includegraphics[width=.25\paperwidth,height=\paperheight]{#3}};
      \node[anchor=south west] at ([xshift=.75\paperwidth]current page.south west)
        {\includegraphics[width=.25\paperwidth,height=\paperheight]{#4}};

      % Fold guides
      \draw[PanelLine,opacity=0.35,line width=0.7pt]
        ([xshift=.25\paperwidth]current page.south west) -- ([xshift=.25\paperwidth]current page.north west);
      \draw[PanelLine,opacity=0.35,line width=0.7pt]
        ([xshift=.50\paperwidth]current page.south west) -- ([xshift=.50\paperwidth]current page.north west);
      \draw[PanelLine,opacity=0.35,line width=0.7pt]
        ([xshift=.75\paperwidth]current page.south west) -- ([xshift=.75\paperwidth]current page.north west);
    \end{tikzpicture}%
  }
}
